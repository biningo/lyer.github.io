x^0=x
x^(~x)=全1 #位数和x一样的全1
x^x=0

交换两个整数变量

a^=b

判断奇偶性

x&1 == 1 #奇数
x&1 == 0 #偶数

计算某个数字二进制位1的个数

int countOne(int num)
{
    int cnt = 0;
    for (int i = 0; i < 8 * sizeof(num); ++i) {
        cnt += num & 1;
        num >>= 1;
    }
    return cnt;
}

mask=11110000

将最低位的1置0

num & (num - 1)

一个数组中其他数都出现了2次，只有一个数出现了1次，找出那个数

int showOnece(vector<int> nums)
{
    int res = 0;
    for (auto num : nums) {
        res ^= num;
    }
    return res;
}
